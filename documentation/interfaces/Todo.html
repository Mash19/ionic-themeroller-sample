import { Injectable } from "@angular/core";

interface Todo {
  id: string;
  title: string;
  description: string;
  value: string;
}

@Injectable()
export class TodoService {
  public todos: Todo[] = [];

  constructor() {
    // Set some test todos
    this.todos = [

        id: "todo-one",
        title: "Todo One",

        title: "Todo Two",
        description: "Do Stuff",
        value: "2"

        id: "todo-three",

}
